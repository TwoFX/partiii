	\item $\mathbb{A}_k^1\to \Spec k$ is not proper. It is separated (we have not proved

		KP_2\colon	\mathbb{A}_k^1\times_{\Spec k} \mathbb{A}_k^1 = \mathbb{A}_k^2 = \Spec k[X] \otimes_k k[Y] &\to \Spec k[t]= \mathbb{A}_k^1\\

		\end{align*}
